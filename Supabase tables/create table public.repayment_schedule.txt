create table public.repayment_schedule (
  id uuid not null default gen_random_uuid (),
  loan_id uuid not null,
  payment_number integer not null,
  due_date date not null,
  scheduled_amount numeric(12, 2) null,
  principal_portion numeric(12, 2) null,
  interest_portion numeric(12, 2) null,
  paid_amount numeric(12, 2) null,
  paid_date date null,
  status character varying(50) null default 'pending'::character varying,
  created_at timestamp without time zone null default now(),
  updated_at timestamp without time zone null default now(),
  amount numeric(10, 2) not null default 0.00,
  constraint repayment_schedule_pkey primary key (id),
  constraint repayment_schedule_loan_id_fkey foreign KEY (loan_id) references loans (id) on delete CASCADE
) TABLESPACE pg_default;

create index IF not exists idx_repayment_schedule_loan_id on public.repayment_schedule using btree (loan_id) TABLESPACE pg_default;
