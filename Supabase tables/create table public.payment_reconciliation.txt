create table public.payment_reconciliation (
  id uuid not null default gen_random_uuid (),
  loan_id uuid not null,
  client_id uuid not null,
  payment_date date not null,
  total_amount numeric(12, 2) not null,
  fees_amount numeric(12, 2) null default 0,
  interest_amount numeric(12, 2) null default 0,
  principal_amount numeric(12, 2) null default 0,
  payment_method character varying(50) null,
  status character varying(50) null default 'completed'::character varying,
  created_at timestamp without time zone null default now(),
  updated_at timestamp without time zone null default now(),
  constraint payment_reconciliation_pkey primary key (id),
  constraint payment_reconciliation_client_id_fkey foreign KEY (client_id) references clients (id) on delete CASCADE,
  constraint payment_reconciliation_loan_id_fkey foreign KEY (loan_id) references loans (id) on delete CASCADE
) TABLESPACE pg_default;

create index IF not exists idx_payment_reconciliation_loan_id on public.payment_reconciliation using btree (loan_id) TABLESPACE pg_default;

create index IF not exists idx_payment_reconciliation_client_id on public.payment_reconciliation using btree (client_id) TABLESPACE pg_default;
